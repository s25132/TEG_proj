import chromadb
from chromadb.config import Settings

chroma_client = chromadb.Client(
    Settings(
        chroma_db_impl="duckdb+parquet",
        persist_directory="./chroma_db",
    )
)

collection = chroma_client.get_or_create_collection("projects_and_cvs")

query = "experienced React developer for e-commerce project"

results = collection.query(
    query_texts=[query],
    n_results=5,
    # bez 'where' = szuka i w projektach, i w CV
)

print(results)
